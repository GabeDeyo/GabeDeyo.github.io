Setup

  • 2 by 3 layout will work best for this project with the project below the heirarchy window.

  • Drag the environment prefab into the scene and reset.

  • Drag the Lights prefab into the scene and reset if you want.
    Drab those up if you want.

  • Create a Quad and reset as "Floor".
    Rotate to (90, 0, 0) and scale to (100, 100, 1);
    Remove the Mesh Renderer component.
    Set the Floors layer to "Floor".

  • Create an Empty "BackgroundMusic".
    Add an audio source to it or drag the "Background Music" audio.
    Uncheck play on awake and Check loop.
    Put volume to 0.1.

Player Character

  • Drag the Player model into the scene and reset.
    Set tag to "Player"

  • In the Animation folder, create an Animator Controller as "PlayerAC".
    Drag onto the player. Should fill in the Controller option.
    Open the PlayerAC. Drag each Player clip from model into Animator.

  • Set Idle clip to default.
    Add "Bool" parameter named "IsWalking".
    Add "Trigger" parameter named "Die".

  • Make transition from Idle -> Move.
    Set condition to IsWalking, and leave as true.
    Make transition from Move -> Idle.
    Set condition to IsWalking, and set as false.
    Make transition from AnyState -> Die
    Set condition to Die.

  • Add Rigidbody to Player.
    Set Drag and Angular Drag to "inf"
    Set position constraints to Y-constrained
    Set rotation constraints to X/Z-constrained

  • Add Capsule Collider to the Player.
    Center at (0.2, 0.6, 0).
    Set radius to (0.5) and height to (1.2)
    Make sure direction is y-axis.

  • Add "Player Hurt" audio to Player.
    Do not play on awake.

  • Add a "PlayerMovement" script to the player. 

                //PlayerMovement
            public float speed = 6f;

            private Vector3 movement;
            private Animator anim;
            private Rigidbody rb;
            private int floorMask;
            private float camRayLength = 100f;

            void Awake(){
                floorMask = LayerMask.GetMask("Floor");
                anim = GetComponent();
                rb = GetComponent();
            }

            void FixedUpdate() {
                float h = Input.GetAxisRaw("Horizontal"); //-1, 0, or 1
                float v = Input.GetAxisRaw("Vertical");

                Move(h, v);
                Turning();
                Animating(h, v);
            }

            void Move(float h, float v) {
                movement.Set(h, 0f, v);

                movement = movement.normalized * speed * Time.deltaTime;

                rb.MovePosition(transform.position + movement);
            }

            void Turning() {
                Ray camRay = Camera.main.ScreenPointToRay(Input.mousePosition);

                RaycastHit floorHit;

                if(Physics.Raycast(camRay, out floorHit, camRayLength, floorMask)) {
                    Vector3 playerToMouse = floorHit.point - transform.position;
                    playerToMouse.y = 0f;

                    Quaternion newRotation = Quaternion.LookRotation(playerToMouse);
                    rb.MoveRotation(newRotation);
                }
            }

            void Animating(float h, float v) {
                bool walking = h!=0f || v!=0f;
                anim.SetBool("IsWalking", walking);
            }

Camera

  • Move camera to (1, 15, -22), and rotate to (30, 0, 0)
    Set to orthographic view with 4.5 orthographic size.
    Set background color to black.

  • Create a "CameraFollow" script and attach to main camera. 

                //CameraFollow
            public Transform target;
            public float smoothing = 5f;

            private Vector3 offset;

            void Start() {
                offset = transform.position - target.position;
            }

            void FixedUpdate() {
                Vector3 targetCamPos = target.position + offset;
                transform.position = Vector3.Lerp(transform.position, targetCamPos, smoothing * Time.deltaTime);
            }

Player Health

  • Make a new "PlayerHealth" scripts on the Player. 

                //PlayerHealth
            public int startingHealth = 100;
            public int currentHealth;
            public Slider healthSlider;
            public Image damageImage;
            public AudioClip deathClip;
            public float flashSpeed = 5f;
            public Color flashColor = new Color(1f, 0f, 0f, 0.1f);

            private Animator anim;
            private AudioSource playerAudio;
            PlayerMovement playerMovement;
            bool isDead;
            bool damaged;

            void Awake() {
                anim = GetComponenet();
                playerAudio = GetComponenet();
                playerMovement = GetComponent();
                currentHealth = startingHealth;
            }

            void Update() {
                if(damaged) {
                    damageImage.color = flashColor;
                } else {
                    damageImage.color = Color.Lerp(damageImage.color, Color.clear, flashSpeed * Time.deltaTime);
                }
                damaged = false;
            }

            public void TakeDamage(int amount) {
                damaged = true;

                currentHealth -= amount;

                heatlthSlider.value = currentHealth;

                playerAudio.Play();

                if(currentHealth <= 0 && !isDead) {
                    Death();
                }
            }

            void Death() {
                isDead = true;

                anim.SetTrigger("Die");

                playerAudio.clip = deathClip;
                playerAudio.Play();

                playerMovement.enabled = false;
            }

  • Make a new "EnemyAttack" script and add to Zombunny. 

                //EnemyAttack
            public float timeBetweenAttacks = 0.5f;
            public int attackDamage = 10;

            private Animator anim;
            private GameObject player;
            private PlayerHealth playerHealth;
            bool playerInRange;
            float timer;

            void Awake() {
                player = GameObject.FindGameObjectWithTag("Player");
                playerHealth = player.GetComponent();
                anim = GetComponent();
            }

            void OnTriggerEnter(Collider other) {
                if(other.gameObject == player) {
                    playerInRange = true;
                }
            }

            void OnTriggerExit(Collider other) {
                if(other.gameObject == player) {
                    playerInRange = false;
                }
            }

            void Update() {
                timer += Time.deltaTime;

                if(timer >= timeBetweenAttacks && playerInRange) {
                    Attack();
                }

                if(playerHealth.currentHealth <= 0) {
                    anim.SetTrigger("PlayerDead");
                }
            }

            void Attack() {
                timer = 0f;

                if(playerHealth.currentHealth > 0) {
                    playerHealth.TakeDamage(attackDamage);
                }
            }

Harming Enemies

  • Create an EnemyHealth script and attach to the Zombunny. 

                //EnemyHealth
            public int startingHealth = 100;
            public int currentHealth;
            public float sinkSpeed = 2.5f;
            public int scoreValue = 10;
            public AudioClip deathClip;

            private Animator anim;
            private AudioSource enemyAudio;
            private ParticlesSystem hitParticles;
            private CapsuleCollider capsuleCollider;
            private bool isDead;
            private bool isSinking;

            void Awake() {
                anim = GetComponent();
                enemyAudio = GetComponent();
                hitParticles = GetComponentInChildren();
                capsuleCollider = GetComponent();

                currentHealth = startingHealth;
            }

            void Update() {
                if(isSinking) {
                    transform.Translate(-Vector3.up * sinkSpeed * Time.deltaTime);
                }
            }

            public void TakeDamage(int amount, Vector3 hitPoint) {
                if(isDead)
                    return;

                enemyAudio.Play();

                currentHealth -= amount;

                hitParticles.transform.position = hitPoint;
                hitParticles.Play();

                if(currentHealth <= 0) {
                    Death();
                }
            }

            void Death() {
                isDead = true;

                capsuleCollider.isTrigger = true;

                anim.SetTrigger("Dead");

                enemyAudio.clip = deathClip;
                enemyAudio.Play();
            }

            public void StartSinking() {
                //Just turning off this Component!, not the gameObject which uses gameObject.SetActive(false);
                GetComponent().enabled = false;
                GetComponent().isKinematic = true;

                isSinking = true;

                Destory(gameObject, 2f);
            }

  • Now we need to add things to EnemyAttack. 

                //EnemyAttack
            public float timeBetweenAttacks = 0.5f;

  • Make a PlayerShooting script and add to GunBarrelEnd. 

                //PlayerShooting
            public int damagePerShot = 20;
            public float timeBetweenBullets = 0.15f;
            public float range = 100f;

            private float timer;
            private Ray shootRay;
            private RaycastHit shootHit;
            private shootableMask;
            private ParticleSystem gunParticles;
            private LineRenderer gunParticles;
            private AudioSource gunAudio;
            private Light gunLight;
            private float effectsDisplayTime = 0.2f;

            void Awake(){
                shootableMask = LayerMask.GetMask("Shootable");
                gunParticles = GetComponent();
                gunLine = GetComponent();
                gunAudio = GetComponent();
                gunLight = GetComponent();
            }

            void Update() {
                timer += Time.deltaTime;

                if(Inpute.GetButton("Fire1") && timer >= timeBetweenBullets) {
                    Shoot();
                }

                if(timer >= timeBetweenBullets * effectsDisplayTime) {
                    DisableEffect();
                }
            }

            public void DisableEffects() {
                gunLine.enabled = false;
                gunLight.enabled = false;
            }

            void Shoot() {
                timer = 0f;

                gunAudio.Play();

                gunLight.enabled = true;

                gunParticles.Stop();
                gunParticles.Play();

                gunLine.enabled = true;
                gunLine.SetPosition(0, transofmr.position);

                shootRay.origin = transform.position;
                shootRay.direction = transform.forward;

                if(Physics.Raycast(shootRay, out shootHit, range, shootableMask)) {
                    EnemyHealth enemyHealth = shootHit.collider.GetComponent();
                    if(enemyHealth != null) {
                        enemyHealth.TakeDamage(damagerPerShot, shootHit.point);
                    }
                    gunLine.SetPosition(1, shootHit.point);
                } else {
                    gunLine.SetPosition(1, shootRay.origin + shootRay.direction * range);
                }
            }

                //EnemyManager
            public PlayerHealth playerHealth;
            public GameObject enemy;
            public float spawnTime = 3f;
            public Transform[] spawnPoints;

            void Start() {
                InvokeRepeating("Spawn", spawnTime, spawnTime);
            }

            void Spawn() {
                if(playerHealth.currentHealth <= 0f) {
                    return;
                }

                int spawnPointIndex = Random.Range(0, spawnPoints.Length);

                Instantiate(enemy, spawnPoints[spawnPointIndex].position, spawnPoints[spawnPointIndex].rotation);
            }

  • Frame again on the HUDCanvas.
    Add an Image as ScreenFader.
    Alt stretch image over the entire canvas.
    Change the color to light blue.

  • Add a Text object to the HUDCanvas as "GameOverText"

    Change widht/height to 300/50
    Change text to Game Over!

  • The lower in the Hierarchy of a canvas, the more in front it is.

    Move to frame 0:20 and create a keyframe for the GameOverText: RectTransform.Scale:
      With play head at 0:00 make scale for all 0

    Make a transition from Empty -> GameOverClip
    Make a trigger parameter called "GameOver"
    Set the condition of the transition to GameOver.

  • Make a new GameOverManager on the HUDCanvas object. 

                //GameOverManager
            public PlayerHealth playerHealth;
            public float restartDelay = 5f;

            private Animator anim;
            private float restartTimer;

            void Awake() {
                anim = GetComponent();
            }

            void Update() {
                if(playerHealth.currentHealth <= 0) {
                    anim.SetTrigger("GameOver");

                    restartTimer += Time.deltaTime;

                    if(restartTimer >= restartDelay) {
                        Application.LoadLevel(Application.loadedLevel);
                    }
                }
            }
